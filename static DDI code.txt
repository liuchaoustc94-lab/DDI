`%||%` <- function(x, y) {
  if (is.null(x) || length(x) == 0 || (length(x) == 1 && is.na(x))) y else x
}

fmt_txt_num <- function(x, digits = 3) {
  if (is.na(x)) "---" else formatC(x, format = "f", digits = digits)
}

fmt_txt_trim <- function(x, digits = 3) {
  if (is.na(x)) return("---")
  sub("\\.?0+$", "", formatC(x, format = "f", digits = digits))
}

is_iv_route <- function(route) {
  identical(route %||% "Oral", "IV")
}

calc_cmax_uM <- function(c_ng_ml, mw) {
  if (is.na(c_ng_ml) || is.na(mw) || mw <= 0) return(NA_real_)
  c_ng_ml / mw
}

calc_unbound <- function(c_uM, fu_p) {
  if (is.na(c_uM) || is.na(fu_p)) return(NA_real_)
  c_uM * fu_p
}

calc_iin_max_total <- function(cmax_uM, dose_mg, fa, fg, ka_min, qh_L_h, mw, route = "Oral") {
  if (is_iv_route(route)) return(cmax_uM)
  if (any(is.na(c(cmax_uM, dose_mg, fa, fg, ka_min, qh_L_h, mw))) || mw <= 0 || qh_L_h <= 0) return(NA_real_)
  dose_umol <- dose_mg * 1000 / mw
  ka_h <- ka_min * 60
  cmax_uM + (fa * fg * dose_umol * ka_h) / qh_L_h
}

calc_iin_max_unbound <- function(cmax_uM, dose_mg, fa, fg, ka_min, qh_L_h, mw, fu_p, route = "Oral") {
  x <- calc_iin_max_total(cmax_uM, dose_mg, fa, fg, ka_min, qh_L_h, mw, route = route)
  if (is.na(x) || is.na(fu_p)) return(NA_real_)
  x * fu_p
}

calc_igut_uM <- function(dose_mg, mw, gut_vol_L = 0.25, route = "Oral") {
  if (is_iv_route(route)) return(NA_real_)
  if (any(is.na(c(dose_mg, mw, gut_vol_L))) || mw <= 0 || gut_vol_L <= 0) return(NA_real_)
  dose_umol <- dose_mg * 1000 / mw
  dose_umol / gut_vol_L
}

calc_i_ent_uM <- function(dose_mg, mw, ka_min, fa, qent_ml_min, route = "Oral") {
  if (is_iv_route(route)) return(NA_real_)
  if (any(is.na(c(dose_mg, mw, ka_min, fa, qent_ml_min))) || mw <= 0 || qent_ml_min <= 0) return(NA_real_)
  dose_umol <- dose_mg * 1000 / mw
  (ka_min * fa * dose_umol / qent_ml_min) * 1000
}

calc_ih_total_uM <- function(cmax_total_uM, dose_mg, mw, ka_min, fa, fg, qh_ml_min, rb, route = "Oral") {
  if (is_iv_route(route)) return(cmax_total_uM)
  if (any(is.na(c(cmax_total_uM, dose_mg, mw, ka_min, fa, fg, qh_ml_min, rb))) || mw <= 0 || qh_ml_min <= 0 || rb <= 0) return(NA_real_)
  dose_umol <- dose_mg * 1000 / mw
  cmax_total_uM + ((ka_min * fa * fg * dose_umol / qh_ml_min) / rb) * 1000
}

calc_ih_unbound_uM <- function(cmax_total_uM, dose_mg, mw, ka_min, fa, fg, qh_ml_min, rb, fu_p, route = "Oral") {
  x <- calc_ih_total_uM(cmax_total_uM, dose_mg, mw, ka_min, fa, fg, qh_ml_min, rb, route = route)
  if (is.na(x) || is.na(fu_p)) return(NA_real_)
  fu_p * x
}

calc_r1 <- function(I, Ki) {
  if (any(is.na(c(I, Ki))) || Ki <= 0) return(NA_real_)
  1 + I / Ki
}

calc_tdi_auc_ratio <- function(I, KI, kinact_min, kdeg_min) {
  if (any(is.na(c(I, KI, kinact_min, kdeg_min))) || KI <= 0 || kdeg_min <= 0) return(NA_real_)
  1 + (kinact_min * I) / (kdeg_min * (KI + I))
}

calc_induction_ratio <- function(I, Emax, EC50) {
  if (any(is.na(c(I, Emax, EC50))) || EC50 <= 0) return(NA_real_)
  1 / (1 + Emax * I / (EC50 + I))
}

calc_cheng_prusoff_ki <- function(ic50, s, km) {
  if (any(is.na(c(ic50, s, km))) || km <= 0) return(NA_real_)
  ic50 / (1 + s / km)
}

calc_transporter_ratio <- function(I, Ki) {
  if (any(is.na(c(I, Ki))) || Ki <= 0) return(NA_real_)
  I / Ki
}

combine_static_terms <- function(...) {
  vals <- c(...)
  vals <- vals[!is.na(vals)]
  if (!length(vals)) return(NA_real_)
  prod(vals)
}

calc_hepatic_net_aucr <- function(effect_term, fm) {
  if (is.na(effect_term) || is.na(fm)) return(NA_real_)
  1 / (effect_term * fm + (1 - fm))
}

calc_gut_net_aucr <- function(effect_term, fg) {
  if (is.na(effect_term) || is.na(fg)) return(NA_real_)
  1 / (effect_term * (1 - fg) + fg)
}

combine_aucr_terms <- function(liver_aucr, gut_aucr) {
  if (is.na(liver_aucr) && is.na(gut_aucr)) return(NA_real_)
  if (is.na(liver_aucr)) return(gut_aucr)
  if (is.na(gut_aucr)) return(liver_aucr)
  liver_aucr * gut_aucr
}

pk_summary <- function(x) {
  cmax_total_uM <- calc_cmax_uM(x$cmax_ng_ml, x$mw_calc)
  cmax_unbound_uM <- calc_unbound(cmax_total_uM, x$fu_p)
  iin_max_total_uM <- calc_iin_max_total(
    cmax_uM = cmax_total_uM,
    dose_mg = x$dose_mg,
    fa = x$fa,
    fg = x$fg,
    ka_min = x$ka_min,
    qh_L_h = x$qh_L_h,
    mw = x$mw_calc,
    route = x$route %||% "Oral"
  )
  iin_max_unbound_uM <- calc_iin_max_unbound(
    cmax_uM = cmax_total_uM,
    dose_mg = x$dose_mg,
    fa = x$fa,
    fg = x$fg,
    ka_min = x$ka_min,
    qh_L_h = x$qh_L_h,
    mw = x$mw_calc,
    fu_p = x$fu_p,
    route = x$route %||% "Oral"
  )
  igut_uM <- calc_igut_uM(
    dose_mg = x$dose_mg,
    mw = x$mw_calc,
    gut_vol_L = x$gut_vol_L %||% 0.25,
    route = x$route %||% "Oral"
  )
  
  list(
    cmax_total_uM = cmax_total_uM,
    cmax_unbound_uM = cmax_unbound_uM,
    iin_max_total_uM = iin_max_total_uM,
    iin_max_unbound_uM = iin_max_unbound_uM,
    igut_uM = igut_uM
  )
}

parse_formula_counts <- function(formula) {
  if (is.null(formula) || !nzchar(trimws(formula))) return(NULL)
  
  s <- gsub("\\s+", "", formula)
  n <- nchar(s)
  pos <- 1
  
  parse_number <- function() {
    if (pos > n) return(1)
    rest <- substr(s, pos, n)
    m <- regexpr("^[0-9]+", rest, perl = TRUE)
    if (m[1] == -1) return(1)
    len <- attr(m, "match.length")
    val <- as.numeric(substr(rest, 1, len))
    pos <<- pos + len
    val
  }
  
  parse_element <- function() {
    if (pos > n) return(NULL)
    rest <- substr(s, pos, n)
    m <- regexpr("^[A-Z][a-z]?", rest, perl = TRUE)
    if (m[1] == -1) return(NULL)
    len <- attr(m, "match.length")
    el <- substr(rest, 1, len)
    pos <<- pos + len
    el
  }
  
  merge_counts <- function(a, b, mult = 1) {
    for (nm in names(b)) {
      a[[nm]] <- (a[[nm]] %||% 0) + b[[nm]] * mult
    }
    a
  }
  
  parse_group <- function(stop_on_right_paren = FALSE) {
    counts <- list()
    
    while (pos <= n) {
      ch <- substr(s, pos, pos)
      
      if (ch == "(") {
        pos <<- pos + 1
        inner <- parse_group(stop_on_right_paren = TRUE)
        if (is.null(inner)) return(NULL)
        mult <- parse_number()
        counts <- merge_counts(counts, inner, mult)
      } else if (ch == ")") {
        if (!stop_on_right_paren) return(NULL)
        pos <<- pos + 1
        return(counts)
      } else {
        el <- parse_element()
        if (is.null(el)) return(NULL)
        mult <- parse_number()
        counts[[el]] <- (counts[[el]] %||% 0) + mult
      }
    }
    
    if (stop_on_right_paren) return(NULL)
    counts
  }
  
  counts <- parse_group(FALSE)
  if (is.null(counts)) return(NULL)
  if (pos <= n) return(NULL)
  counts
}

calc_mw_from_formula <- function(formula) {
  counts <- parse_formula_counts(formula)
  if (is.null(counts)) return(NA_real_)
  
  total <- 0
  for (el in names(counts)) {
    if (!(el %in% names(atomic_weights))) return(NA_real_)
    total <- total + atomic_weights[[el]] * counts[[el]]
  }
  total
}

transporter_formula_html <- function(site, route = "Oral") {
  if (identical(site, "gut")) {
    if (is_iv_route(route)) {
      return(HTML("<span class='tr-report-formula'>Not applicable for i.v.</span>"))
    }
    return(HTML("<span class='tr-report-formula'>[Dose 250mL] / K<sub>i,u</sub></span>"))
  }
  if (identical(site, "hepatic_inlet")) {
    return(HTML("<span class='tr-report-formula'>I<sub>in,max</sub> / K<sub>i,u</sub></span>"))
  }
  if (identical(site, "hepatic_inlet_unbound")) {
    return(HTML("<span class='tr-report-formula'>I<sub>in,max,u</sub> / K<sub>i,u</sub></span>"))
  }
  HTML("<span class='tr-report-formula'>C<sub>max,u</sub> / K<sub>i,u</sub></span>")
}

}

label_html <- function(x) div(class = "field-label", HTML(x))
note_html <- function(x) div(class = "field-note", HTML(x))
reset_button <- function(id) actionButton(id, label = HTML("&#8634;"), class = "reset-btn")

input_num <- function(id, value = NA, step = NULL, min = NULL, max = NULL, width = "100%") {
  args <- list(inputId = id, label = NULL, value = value, width = width)
  if (!is.null(step)) args$step <- step
  if (!is.null(min)) args$min <- min
  if (!is.null(max)) args$max <- max
  do.call(numericInput, args)
}

field_pair_numeric <- function(id, label, value = NA, ref = "", note = NULL, step = NULL, min = NULL, max = NULL) {
  div(
    class = "pair-block",
    div(
      class = "pair-head",
      div(class = "pair-left-head", label_html(label)),
      div(class = "pair-right-head", label_html("Reference")),
      div()
    ),
    div(
      class = "pair-row",
      div(class = "pair-left", input_num(id, value = value, step = step, min = min, max = max)),
      div(class = "pair-right", textInput(paste0(id, "_ref"), NULL, value = ref, width = "100%")),
      div(class = "pair-reset", reset_button(paste0("reset_", id)))
    ),
    if (!is.null(note)) note_html(note)
  )
}

field_pair_select <- function(id, label, choices, selected = NULL, ref = "", note = NULL) {
  div(
    class = "pair-block",
    div(
      class = "pair-head",
      div(class = "pair-left-head", label_html(label)),
      div(class = "pair-right-head", label_html("Reference")),
      div()
    ),
    div(
      class = "pair-row",
      div(class = "pair-left", selectInput(id, NULL, choices = choices, selected = selected, width = "100%")),
      div(class = "pair-right", textInput(paste0(id, "_ref"), NULL, value = ref, width = "100%")),
      div(class = "pair-reset", reset_button(paste0("reset_", id)))
    ),
    if (!is.null(note)) note_html(note)
  )
}

}

server <- function(input, output, session) {
  default_formula <- "C19H22F3N5O2S"
  default_mw <- calc_mw_from_formula(default_formula)
  
  chem_state <- reactiveValues(
    drug_name = "NVP-BYL719",
    formula = default_formula,
    mim = paste0(fmt_txt_num(default_mw, 2), " [Da]"),
    mam = paste0(fmt_txt_num(default_mw, 2), " [Da]"),
    avg = paste0(fmt_txt_num(default_mw, 2), " [Da]"),
    mw_value = default_mw
  )
  
  output$page_title <- renderText({
    if (identical(input$main_tabs, "system")) {
      "System input parameters"
    } else if (identical(input$main_tabs, "intermediate")) {
      "Intermediate parameter values"
    } else if (identical(input$main_tabs, "report")) {
      "Cyp DDI risk assessment report"
    } else if (identical(input$main_tabs, "tr_report")) {
      "Transporter DDI risk assessment report"
    } else {
      "Substance input parameters"
    }
  })
  
  output$report_title_txt <- renderText(paste0("Cyp DDI risk assessment report for ", chem_state$drug_name))
  output$tr_report_title_txt <- renderText(paste0("Transporter DDI risk assessment report for ", chem_state$drug_name))
  
  output$drug_name_txt <- renderText(chem_state$drug_name)
  output$formula_txt <- renderText(chem_state$formula)
  output$mim_txt <- renderText(chem_state$mim)
  output$mam_txt <- renderText(chem_state$mam)
  output$avg_txt <- renderText(chem_state$avg)
  
  output$structure_ui <- renderUI({
    div(
      class = "structure-placeholder",
      "Chemical structure preview removed"
    )
  })
  
  observe({

    if (!nzchar(formula)) return(NULL)
    
    mw <- calc_mw_from_formula(formula)
    chem_state$formula <- formula
    
    if (!is.na(mw)) {
      chem_state$mim <- paste0(fmt_txt_num(mw, 2), " [Da]")
      chem_state$mam <- paste0(fmt_txt_num(mw, 2), " [Da]")
      chem_state$avg <- paste0(fmt_txt_num(mw, 2), " [Da]")
      chem_state$mw_value <- mw
      
      updateSelectInput(
        session,
        "mw_display",
        choices = c(paste0(fmt_txt_num(mw, 2), " (AVG)")),
        selected = paste0(fmt_txt_num(mw, 2), " (AVG)")
      )
    } else {
      chem_state$mim <- "---"
      chem_state$mam <- "---"
      chem_state$avg <- "---"
      chem_state$mw_value <- NA_real_
      
      updateSelectInput(
        session,

        choices = c("---"),

      )

  })
  
  observeEvent(input$reset_formula_input, {

      choices = c(paste0(fmt_txt_num(default_mw, 2), " (AVG)")),

  observeEvent(input$reset_mw_display, {

        updateNumericInput(session, paste0("opt_tr_km_", ii), value = NA_real_)

  output$system_cyp_ui <- renderUI({
    tagList(lapply(seq_len(nrow(system_cyp)), function(i) {

      "OCT2" = HTML("<span class='tr-report-formula'>C<sub>max,u</sub> / K<sub>i,u</sub></span>"),
      "MRP2" = HTML("<span class='tr-report-formula'>I<sub>in,max,u</sub> / K<sub>i,u</sub></span>"),
      "MATE1" = HTML("<span class='tr-report-formula'>C<sub>max,u</sub> / K<sub>i,u</sub></span>"),
      "MATE2K" = HTML("<span class='tr-report-formula'>C<sub>max,u</sub> / K<sub>i,u</sub></span>"),
      "BSEP" = HTML("<span class='tr-report-formula'>I<sub>in,max,u</sub> / K<sub>i,u</sub></span>"),
      "NTCP" = HTML("<span class='tr-report-formula'>I<sub>in,max,u</sub> / K<sub>i,u</sub></span>")
    )
    
    calc_I_value <- function(site) {
      if (site == "gut") return(igut)
      if (site == "hepatic_inlet") return(ih_total)
      if (site == "hepatic_inlet_unbound") return(ih_unbound)
      if (site == "systemic_unbound") return(pk$cmax_unbound_uM)
      pk$cmax_unbound_uM
    }
    
    transporter_rows <- lapply(seq_len(nrow(tr_df)), function(i) {
      nm <- tr_df$transporter[i]
      site <- tr_df$site[i]
      ki_val <- tr_df$Ki_uM[i]
      if (is.na(ki_val)) ki_val <- calc_cheng_prusoff_ki(tr_df$IC50_uM[i], tr_df$S_uM[i], tr_df$Km_uM[i])
      oral_gut_only <- nm %in% c("P-gp", "BCRP") && !is_iv_route(input$route)
      iv_pgp_bcrp <- nm %in% c("P-gp", "BCRP") && is_iv_route(input$route)
      i_value <- if (oral_gut_only) {
        igut
      } else if (iv_pgp_bcrp) {
        pk$cmax_unbound_uM
      } else {
        calc_I_value(site)
      }
      ratio <- calc_transporter_ratio(i_value, ki_val)
      threshold_num <- if (iv_pgp_bcrp) {
        0.02
      } else {
        switch(
          nm,
          "P-gp" = 10,
          "BCRP" = 10,
          "MATE1" = 0.02,
          "MATE2K" = 0.02,
          "BSEP" = 0.02,
          0.1
        )
      }
      formula_html <- if (iv_pgp_bcrp) {
        HTML("<span class='tr-report-formula'>C<sub>max,u</sub> / K<sub>i,u</sub></span>")
      } else {
        formula_map[[nm]]
      }
      threshold_txt <- if (iv_pgp_bcrp) {
        "&ge;0.02"
      } else {
        threshold_map[[nm]]
      }
      organ_html <- if (iv_pgp_bcrp) "Systemic *" else organ_map[[nm]]
      label_txt <- if (iv_pgp_bcrp) nm else tr_label_map[[nm]]
      red_flag <- !is.na(ratio) && ratio >= threshold_num
      
      tags$tr(
        tags$td(HTML(organ_html)),
        tags$td(label_txt),
        tags$td(class = "tr-report-center tr-report-mono", fmt_txt_num(ki_val)),
        tags$td(formula_html),
        tags$td(class = if (red_flag) "tr-report-red tr-report-center tr-report-mono" else "tr-report-center tr-report-mono", fmt_txt_num(ratio)),
        tags$td(class = "tr-report-center", HTML(threshold_txt))
      )
    })
    
    optional_rows <- lapply(seq_len(nrow(opt_tr_df)), function(i) {
      nm <- opt_tr_df$transporter[i]
      if (!nzchar(nm)) nm <- "---"
      site <- opt_tr_df$site[i]
      ki_val <- opt_tr_df$Ki_uM[i]
      if (is.na(ki_val)) ki_val <- calc_cheng_prusoff_ki(opt_tr_df$IC50_uM[i], opt_tr_df$S_uM[i], opt_tr_df$Km_uM[i])
      i_value <- calc_I_value(site)
      ratio <- calc_transporter_ratio(i_value, ki_val)
      threshold_num <- opt_tr_df$threshold[i]
      red_flag <- !is.na(ratio) && !is.na(threshold_num) && ratio >= threshold_num
      tags$tr(
        tags$td(HTML(transporter_organ_html(site, input$route))),
        tags$td(nm),
        tags$td(class = "tr-report-center tr-report-mono", fmt_txt_num(ki_val)),
        tags$td(transporter_formula_html(site, input$route)),
        tags$td(class = if (red_flag) "tr-report-red tr-report-center tr-report-mono" else "tr-report-center tr-report-mono", fmt_txt_num(ratio)),
        tags$td(class = "tr-report-center", HTML(paste0("&ge;", fmt_txt_trim(threshold_num))))
      )
    })
    
    tags$table(
      class = "tr-report-table tr-report-tight",
      style = "width: 900px;",
      tags$thead(
        tags$tr(
          tags$th("Organ"),
          tags$th("Transporter"),
          tags$th(HTML("K<sub>i,u</sub> (&mu;M)")),
          tags$th("Formula"),
          tags$th(HTML("[I] / K<sub>i,u</sub>")),
          tags$th(HTML("ICH M12<br>Threshold"))
        )
      ),
      tags$tbody(
        transporter_rows,
        tags$tr(tags$td(colspan = 6, class = "tr-report-center", "Optional Transporters")),
        optional_rows,
        tags$tr(
          tags$td(colspan = 6, HTML("<span class='tr-report-red' style='padding:0 4px;'>Red background</span> means \"there's a risk of DDI\""))
        ),
        tags$tr(
          tags$td(colspan = 6, HTML("* For oral P-gp/BCRP use [Dose/250 mL]/K<sub>i,u</sub>; for i.v. dosing use C<sub>max,u</sub>/K<sub>i,u</sub> with the post-absorption threshold."))
        )
      )
    )
  })
}
